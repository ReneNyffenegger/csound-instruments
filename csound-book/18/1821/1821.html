instr 1821, 1822
1821.orc
1821.sco
  instr 1821 ; context sensitive global rms
igm = 1.618
imp = p4
kln line 3, p3, 0
afm1 foscil imp, p5, 1, p5*igm, kln, 1
asig linen afm1, 0.01*p3, p3, 0.6*p3
  out asig
gkpwr rms asig
  endin  
 
  instr 1822 ; pulsed noise
klfo oscil 0.5, 8, 1
klfo1 = klfo+0.5 ; rescale [0-1]
anoiz rand 1
apulse = anoiz*klfo1
kpw = p4-gkpwr ; power curve
kmp port kpw, 0.2 ; smooth curve
asig = apulse*kmp
  out asig
  endin
