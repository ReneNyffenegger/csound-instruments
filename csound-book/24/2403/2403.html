  instr 2403 ; Medium Room Reverb

iamp = p4
iinch = p5
adel71 init 0
adel11 init 0
adel12 init 0
adel13 init 0
adel31 init 0
adel61 init 0
adel62 init 0

aflt02 butterbp .4*adel71, 1000, 500 ; Feedback Filter

asum11 = adel12-.35*adel11 ; First Inner Feedforward

; Combine Outputs
