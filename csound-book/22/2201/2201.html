  instr 2201 ; pluck with echoes
kgate linseg p4, .1, 0, 1, 0 ; .1 sec gate
icps = cpspch(p5) ; p4=amp, p5=pitch
asig pluck kgate, icps, icps, 0, 1, 0, 0 ; p6 & p7 = Delay in Sec
atap1 delay asig*.7, p6 ; Make two slapback echoes,
atap2 delay asig*.5, p7 ; ... each one softer
  outs asig+atap1, asig+atap2 ; add stereo echoes
  endin
